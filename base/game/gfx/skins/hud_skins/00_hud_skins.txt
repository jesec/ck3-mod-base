norse = {
	is_valid = {
		has_dlc_feature = the_northern_lords
	}
	is_shown = {
		OR = {
			has_culture = culture:norse
			AND = {
				religion = religion:germanic_religion
				culture = { has_cultural_pillar = heritage_north_germanic }
			}
		}
	}
}

iberian = {
	is_valid = {
		has_dlc_feature = the_fate_of_iberia
	}
	is_shown = {
		culture = { has_cultural_pillar = heritage_iberian }
	}
}

iranian = {
	is_valid = {
		has_dlc_feature = legacy_of_persia
	}
	is_shown = {
		OR = {
			culture = { has_cultural_pillar = heritage_iranian }
			AND = {
				culture = { has_cultural_pillar = heritage_turkic }
				NOT = {
					government_has_flag = government_is_nomadic
				}
			}
		}
	}
}

byzantine = {
	is_valid = {
		has_dlc_feature = roads_to_power
	}
	is_shown = {
		OR = {
			culture = { has_cultural_pillar = heritage_byzantine }
			culture = { has_cultural_pillar = heritage_caucasian }
		}
	}
}

nomad = {
	is_valid = {
		has_dlc_feature = khans_of_the_steppe
	}
	is_shown = {
		government_has_flag = government_is_nomadic
	}
}

asia = {
	is_shown = {
		has_tgp_dlc_trigger = yes
		OR = {
			culture = { has_cultural_pillar = heritage_chinese }
			culture = { has_cultural_pillar = heritage_japonic }
			culture = { has_cultural_pillar = heritage_indo_aryan }
			culture = { has_cultural_pillar = heritage_tai }
			culture = { has_cultural_pillar = heritage_mon_khmer }
			culture = { has_cultural_pillar = heritage_austronesian }
			culture = { has_cultural_pillar = heritage_korean }
			culture = { has_cultural_pillar = heritage_tungusic }
			culture = { has_cultural_pillar = heritage_ainuic }
			culture = { has_cultural_pillar = heritage_qiangic }
			culture = { has_cultural_pillar = heritage_buyeo }
			culture = { has_cultural_pillar = heritage_tibetan }
			culture = { has_cultural_pillar = heritage_viet }
			culture = { has_cultural_pillar = heritage_hmongic }
			culture = { has_cultural_pillar = heritage_malayic }
			culture = { has_cultural_pillar = heritage_nivkh }
			culture = { has_cultural_pillar = heritage_burman }
			culture = { has_cultural_pillar = heritage_dravidian }
			AND = {
				culture = { has_cultural_pillar = heritage_mongolic }
				NOT = {
					government_has_flag = government_is_nomadic
				}
			}
		}
	}
}
