norse = {
	is_valid = {
		has_dlc_feature = the_northern_lords
	}
	is_shown = {
		OR = {
			has_culture = culture:norse
			AND = {
				religion = religion:germanic_religion
				culture = { has_cultural_pillar = heritage_north_germanic }
			}
		}
	}
}
