macro = {
	description = "Is the character a Nomad?"
	definition = "IsNomad(TargetCharacter)"
	replace_with = "And( TargetCharacter.GetGovernment.IsType( 'nomad_government' ), TargetCharacter.GetDomicile.GetType.IsDomicileType('yurt') )"
}

macro = {
	description = "Is the character a Herder?"
	definition = "IsHerder(TargetCharacter)"
	replace_with = "TargetCharacter.GetGovernment.IsType( 'herder_government' ) )"
}
